# 言（ことば）言語仕様 v1.0 — 構文仕様（完全正式文法）

> 状態: Draft  
> バージョン: v1.0  
> 関連規範 ID: SYN-001..006

---

## 3.1 構文方針

**[規範] SYN-001 (MUST)**
`助詞式`、`束縛式`、`アクセス式` は文法規則を分離して定義する。

**[規範] SYN-002 (MUST)**
`返す`、`試す`、`使う`、`公開` は文法上の独立規則を持つ。

---

## 3.2 EBNF

```ebnf
program            ::= sep* statement (sep+ statement)* sep* EOF ;
sep                ::= NEWLINE | "。" ;

statement          ::= public_decl
                     | mutable_bind_stmt
                     | bind_stmt
                     | rebind_stmt
                     | return_stmt
                     | continue_stmt
                     | break_stmt
                     | proc_def
                     | struct_def
                     | trait_def
                     | trait_impl
                     | use_stmt
                     | expr_stmt ;

public_decl        ::= "公開" (proc_def | struct_def | trait_def) ;

mutable_bind_stmt  ::= "変わる" bind_target "は" expr ;
bind_stmt          ::= bind_target "は" expr ;
bind_target        ::= identifier | destructure_pattern ;
rebind_stmt        ::= (identifier | access_expr) "を" expr "に" "変える" ;

return_stmt        ::= "返す" expr? ;
continue_stmt      ::= "次へ" ;
break_stmt         ::= "抜ける" ;

proc_def           ::= identifier "という" "手順" param_decl? block ;
param_decl         ::= "【" param ("、" param)* "】" ;
param              ::= identifier ":" role_particle ;

struct_def         ::= identifier "という" "組" INDENT field_decl+ DEDENT ;
field_decl         ::= identifier "は" identifier ;
trait_def          ::= identifier "という" "特性" INDENT method_sig+ DEDENT ;
method_sig         ::= identifier "という" "手順" param_decl? ("→" identifier)? ;
trait_impl         ::= identifier "は" identifier "を" "持つ" INDENT proc_def+ DEDENT ;

use_stmt           ::= string_lit "を" "使う"
                     | string_lit "から" import_items "を" "使う" ;
import_items       ::= string_lit ("、" string_lit)* ;

expr_stmt          ::= expr ;

expr               ::= te_chain_expr | logical_or_expr ;

te_chain_expr      ::= te_head "、" te_tail ;
te_head            ::= te_step_te ;
te_tail            ::= te_segment ("、" te_segment)* ;
te_segment         ::= te_step_te | te_step_stop | te_branch ;
te_step_te         ::= chain_arg* verb_te ;
te_step_stop       ::= chain_arg* verb_stop ;
chain_arg          ::= expr role_particle_non_wo ;
te_branch          ::= "分岐して" block ;

logical_or_expr    ::= logical_and_expr ("または" logical_and_expr)* ;
logical_and_expr   ::= unary_not_expr ("かつ" unary_not_expr)* ;
unary_not_expr     ::= comparison_expr ("でない")? ;

comparison_expr    ::= additive_expr
                     | additive_expr "が" additive_expr "より" "大きい"
                     | additive_expr "が" additive_expr "より" "小さい"
                     | additive_expr "が" additive_expr "以上"
                     | additive_expr "が" additive_expr "以下"
                     | additive_expr "が" additive_expr "と" "等しい"
                     | additive_expr "が" additive_expr "と" "等しくない" ;

additive_expr      ::= multiplicative_expr
                     | multiplicative_expr "と" multiplicative_expr "の" "和"
                     | multiplicative_expr "と" multiplicative_expr "の" "差" ;

multiplicative_expr ::= postfix_expr
                      | postfix_expr "と" postfix_expr "の" "積"
                      | postfix_expr "を" postfix_expr "で" "割る"
                      | postfix_expr "を" postfix_expr "で" "割った余り" ;

postfix_expr       ::= particle_call_expr | access_expr ;
particle_call_expr ::= particle_arg+ verb_stop ;
particle_arg       ::= access_expr role_particle ;

access_expr        ::= primary_expr ("の" access_tail)* ;
access_tail        ::= identifier
                     | integer_lit "番目"
                     | "長さ" ;

primary_expr       ::= literal
                     | identifier
                     | kosoado_ref
                     | "こう"
                     | list_lit
                     | map_lit
                     | struct_lit
                     | lambda_expr
                     | if_expr
                     | match_expr
                     | loop_expr
                     | try_expr
                     | "(" expr ")" ;
struct_lit         ::= identifier "を" "作る" "【" field_init ("、" field_init)* "】" ;
field_init         ::= identifier ":" expr ;

kosoado_ref        ::= ("これ" | "それ" | "あれ") "の" identifier ;

if_expr            ::= "もし" expr "ならば" block
                       ("もしくは" expr "ならば" block)*
                       ("そうでなければ" block)? ;

match_expr         ::= expr "は" "どれかを調べる" match_block ;
match_block        ::= INDENT match_arm+ DEDENT ;
match_arm          ::= pattern "の場合" block
                     | "どれでもない場合" block ;

pattern            ::= literal
                     | identifier
                     | "どれか"
                     | "【" pattern ("、" pattern)* "】" ;

loop_expr          ::= times_loop_expr
                     | range_loop_expr
                     | while_loop_expr
                     | foreach_loop_expr ;

times_loop_expr    ::= expr "回" "繰り返す" loop_var? block ;
range_loop_expr    ::= expr "から" expr "まで" "繰り返す" loop_var? block ;
while_loop_expr    ::= expr "間" "繰り返す" block ;
foreach_loop_expr  ::= expr "の" "それぞれについて" loop_var block ;
loop_var           ::= "【" identifier "】" ;

try_expr           ::= "試す" block catch_clause? finally_clause? ;
catch_clause       ::= "失敗した場合" catch_param? block ;
catch_param        ::= "【" identifier ":" "で" "】" ;
finally_clause     ::= "必ず行う" block ;

lambda_expr        ::= "（" param_decl? lambda_body "）" ;
lambda_body        ::= statement (sep+ statement)* ;

list_lit           ::= "【" (expr ("、" expr)*)? "】" ;
map_lit            ::= "｛" (map_entry ("、" map_entry)*)? "｝" ;
map_entry          ::= identifier ":" expr ;

destructure_pattern ::= list_destructure | map_destructure ;
list_destructure   ::= "【" identifier "、" identifier "…" "】" ;
map_destructure    ::= "｛" identifier ("、" identifier)* "｝" ;

block              ::= INDENT statement (sep+ statement)* sep* DEDENT ;

literal            ::= integer_lit | float_lit | string_lit | bool_lit | none_lit | counter_lit ;
counter_lit        ::= (integer_lit | float_lit) counter_word ;

role_particle      ::= "を" | "が" | "に" | "で" | "から" | "まで" | "と" | "へ" | "より" ;
role_particle_non_wo ::= "が" | "に" | "で" | "から" | "まで" | "と" | "へ" | "より" ;

verb_stop          ::= identifier ;
verb_te            ::= identifier_te ;
```

---

## 3.3 構文制約

**[規範] SYN-003 (MUST NOT)**
`【:を】` のような無名仮引数宣言を禁止する。仮引数は必ず `名前:助詞` 形式とする。

**[規範] SYN-004 (MUST NOT)**
`返す` を手順名にしてはならない。

**[規範] SYN-005 (MUST)**
`公開` は `手順`・`組`・`特性` の定義にのみ適用できる。

**[規範] SYN-006 (MUST)**
`かつ` は `または` より高優先順位とする。
